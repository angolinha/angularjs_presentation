Dependency injection

myapp/controllers.js 
angular.module('myapp')
  .controller('UserController', function($scope, $timeout){
    $timeout(function() {
      $scope.username = "Charles Montgomery Burns";
    }, 3000);
  });
Pozdraviť:

Hello {{username}}!

Scope

Scope

<div ng-controller="UserController">
  <button ng-click="setUsername()">Set username</button>
  <button ng-click="setUsername('Maggie')">Set Maggie</button>
  <input ng-model="username">
  Hello {{username}}
</div>
angular.module('myapp')
  .controller('UserController', funtion($scope){
    $scope.setUsername = function(username){
      $scope.username = username ? username : "Lisa";
    }
  });

Direktívy & Filtre

Direktívy

<html ng-app="myapp">
  <body ng-init="workers=['Homer','Lenny','Carl']">
    <ul>
      <li ng-repeat="worker in workers"> {{worker}} <li>
    </ul>
    <script src="js/angular.js"></script>
  </body>
</html>

Sector 7G workers:

{{worker}}

Filtre

<html ng-app="myapp">
 <body>
  <input type="text" ng-model="filterName">
  <ul>
   <li ng-repeat="worker in workers | filter:filterName
          | orderBy:'name'">
    {{worker.name}} - {{worker.position}}<li>
  </ul>
 </body>
</html>
Pracovník:
{{worker.name | uppercase}} - {{worker.position}}
