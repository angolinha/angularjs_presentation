Controllers, Scope & Dependency Injection

<!doctype html>
<html ng-app="myapp">
  ...
  <body>
    <div ng-controller="UserController">
      <input ng-model="username">
      Hello {{username}}
    </div>
    
    <script src="js/angular.js"></script>
    <script src="myapp/main.js"></script>
    <script src="myapp/controllers.js"></script>
  </body>
</html>
myapp/controllers.js 
angular.module('myapp')
  .controller('UserController', function($scope) {
    $scope.username = "Homer Simpson";
});

angular.module('myapp')
  .controller('UserController', funtion($scope){
    $scope.setUsername = function(username){
      $scope.username = username ? username : "Lisa";
    }
  });
